# --
# emulator

# set target
idf.py set-target esp32-S3

# build
idf.py build

# flash
idf.py flash -p /dev/ttyUSB0

# monitor
idf.py monitor -p /dev/ttyUSB0

# run with emulator qemu: 
# https://docs.espressif.com/projects/esp-idf/en/stable/esp32s3/api-guides/tools/qemu.html

# qemu monitor
idf.py qemu monitor
